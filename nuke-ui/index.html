nuke的界面

nuke.message("this is message") 提示框

nuke.ask("create a node?") 询问框，点yes返回True

nuke.getInput("input:", "test..") 输入框，返回值为输入的内容，第二个参数为默认值

nuke.getColor() 颜色拾取框，返回整数形式的颜色

nuke.getFilename("get file", "*.nk") 文件浏览器

nuke.getClipname("get clip") 文件选择，可以支持选择序列文件夹

nuke.panel("test panel") 一个面板窗口

关于面板窗口的例子

p = nuke.Panel("test panel")
p.addClipnameSearch("clip path", "/tmp")
p.addFilenameSearch("file path", "/tmp")
p.addTextFontPulldown("font browser", "/myFonts/")
p.addRGBColorChip("some pretty color", "")
p.addExpressionInput("enter an expression", "4*25")
p.addBooleanCheckBox("Yes or no?", True)
p.addEnumerationPulldown("my choices", "A B C")
p.addScriptCommand("tcl or python code", "")
p.addSingleLineInput("just one line", "not much space")
p.addMultilineTextInput("multiple lines of user input", "lineA\\nlineB")
p.addPasswordInput("password", "donttellanyone")
p.addButton("push here")
ret = p.show()

用dot节点画圆的例子

import nuke
import math

def main():
  circleType = "circle spiral both"
  circleNum = "50"
  scaleCircle = "50"
  spiralNum = "100"
  line = "5"
  scaleSpiral = "0.5"
  rotateOffsetX = "0"
  rotateOffsetY = "0"
  clockwise = True
  connectDots = False
  hideinput = True

  p = nuke.Panel("draw circle")
  p.addEnumerationPulldown("circle type:", circleType)
  p.addSingleLineInput("circle num:", circleNum)
  p.addSingleLineInput("circle scale:", scaleCircle)

  p.addSingleLineInput("spiral num:", spiralNum)
  p.addSingleLineInput("spiral scale:", scaleSpiral)
  p.addSingleLineInput("line num:", line)
  p.addSingleLineInput("rotateOffsetX:", rotateOffsetX)
  p.addSingleLineInput("rotateOffsetY:", rotateOffsetY)
  p.addBooleanCheckBox("clockwise:", clockwise)

  p.addBooleanCheckBox("connect dots:", connectDots)
  p.addBooleanCheckBox("hide input:", hideinput)

  result = p.show()
  if result == 1:
    circleType = p.value("circle type:")
    if circleType == "circle" or circleType == "both":
      circleNum = int(p.value("circle num:"))
      scaleCircle  float(p.value("circle scale:"))
    if circleType = "spiral" or circleType == "both":
      spiralNum = int(p.value("spiral num:"))
      scaleSpiral = float(p.value("spiral scale:"))
      line = int(p.value("line num:"))
      rotateOffsetX = float(p.value("rotateOffsetX:"))
      rotateOffsetY = float(p.value("rotateOffsetY:"))
      clockwise = p.value("clockwise")
    connectDots = p.value("connect dots:")
    hideinput = p.value("hide input:")

    if circleType == "circle" or circleType == "both":
      color1 = nuke.getColor()
      for i in range(circleNum):
        x = int(math.cos(2*math.pi / circleNum * i) * scaleCircle)
        y = int(math.sin(2*math.pi / circleNum * i) * scaleCircle)
        dot = nuke.createNode("Dot", inpanel=False)
        dot["xpos"].setValue(x)
        dot["ypos"].setValue(y)
        dot["tile_color"].setValue(color1)
        dot["hide_input"].setValue(hideinput)
        if i == circleNum - 1:
          finalDot = dot
        dot.setSelected(connectDots)
      finalDot.setSelected(False)
    if circleType == "spiral" or circleType == "both":
      color2 = nuke.getColor()
      for i in range(spiralNum):
        if clockwise:
          x = int(-1 * math.cos(i * line + rotateOffsetX * math.pi / 180) * i * scaleSpiral)
        else:
          x = int(1 * math.cos(i * line + rotateOffsetX * math.pi / 180) * i * scaleSpiral)
        y = int(-1 * math.cos(i * line + rotateOffsetY * math.pi / 180) * i * scaleSpiral)
        dot = nuke.createNode("Dot", inpanel=False)
        dot["xpos"].setValue(x)
        dot["ypos"].setValue(y)
