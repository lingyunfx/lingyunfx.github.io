将常量集中到一个文件

# 将常量写在 constant.py
class _const:

    class ConstError(TypeError):
        pass

    class ConstCaseError(ConstError):
        pass

    def __setattr__(self, name, value):
        if name in self.__dict__.keys():
            raise self.ConstError, "Can't change const.{}".format(name)
        if not name.isupper():
            raise self.ConstCaseError, "const name {} is not all uppercase".format(name)
        self.__dict__[name] = value

const = _const()
const.MY_CONSTANT = 1
const.MY_SECOND_CONSTANT = 2
const.MY_THIRD_CONSTANT = 3

# 调用常量
from constant import const

print const.MY_CONSTANT
print const.MY_CONSTANT
print const.MY_SECOND_CONSTANT
print const.MY_THIRD_CONSTANT

数据交换不推荐使用中间变量

# 下面是交换两个变量常用的方法
temp = x
x = y
y = temp

# 更pythonic的方式，也是更好的性能
x, y = y, x

不推荐使用type来进行类型检查

# 内建类型用type检查问题不大，不过建议用isinstance进行类型判断

# 判断2是否为浮点数
isinstance(2, float)

# 判断 'zly' 是否为unicode字符串
isinstance('zly', (str, unicode))

# 判断(1, 5) 是否是str list tuple类型中的一种

异常处理的几点基本原则

try:
	<statements>
# 当try中发生了name1的异常处理
except <name1>:
	<statements>
# 当try中发生了name2或name3中的异常处理
except (name2, name3):
	<statements>
# 当try中发生name4的异常时处理，并获取对应的实例
except <name4> as <data>:
	<statements>
# 其它异常处理
except:
	<statements>
# 没有异常发生时处理
else:
	<statements>
# 不管有没有异常都会执行
finally:
	<statements>
  • 不要在try中放入过多代码
  • except里最好定位具体的异常
  • 注意异常的捕获顺序
  • 使用友好的异常提示信息

Sort和Sorted的区别

sort() 只对列表有效，且会修改列表的排序，函数返回值为None

sorted() 对所有可迭代对象都有效，不会修改原对象，返回修改后的对象。

一些使用的例子

  • 对字典进行排序
from operator import itemgetter
phonebook = {'Linda': '6650', 'Bob': '9545', 'Carol': '5834'}
sorted_pb = sorted(phonebook.iteritems(), key=itemgetter(1))
print sorted_pb
>>> [('Carol', '5834'), ('Linda', '7750'), ('Bob', '9345')]

# 关于itemgetter的用法
a = [1, 2, 3, 4]
get_1 = itemgetter(1)
print get_1(a)
>>> 2

get_23 = itemgetter(2,3)
print get_23(a)
>>> (3, 4)
  • 多维list排序
from operator import itemgetter
gameresult = [['Bob', 95.00, 'A'], ['Alan', 86.0, 'C'], ['Mandy', 82.5, 'A']]
sorted(gameresult, key=itemgetter(2, 1))
>>> [['Mandy', 82.5, 'A'], ['Bob', 95.0, 'A'], ['Alan', 86.0, 'C']]
  • 字典中混合list排序
from operator import itemgetter
mydict = {'Li': ['M', 7], 'Zhang': ['E', 2], 'Wang': ['P', 3]}
sorted(mydict.iteritems(), key=lambda(k, v): itemgetter(1)(v))
>>> [('Zhang', ['E', 2]), ('Wang', ['P', 3]), ('Li', ['M', 7])]
  • list中混合字典排序
gameresult = [{'name': 'Bob', 'wins': 10, 'losses': 3}, {'name': 'Zhang', 'wins': 3, 'losses': 1}, {'name': 'David', 'wins': 12, 'losses': 16}]
sorted(gameresult, key=itemgetter('wins', 'name'))
>>> [{'wins': 3, 'losses': 1, 'name': 'Zhang'}, {'wins': 10, 'losses': 3, 'name': 'Bob'}, {'wins': 12, 'losses': 16, 'name': 'David'}]

使用Counter进行计数统计

使用dict

some_data = ['a', '2', 2, 4, 5, '2', 'b', 4, 7, 'a', 5, 'd', 'a', 'z']
count_frq = dict()
for item in some_data:
	if item in count_frq:
		count_frq[item] += 1
	else:
		count_frq[item] = 1
print count_frq
>>> {'a': 3, 2: 1, 'b': 1, 4: 2, 5: 2, 7: 1, '2': 2, 'z': 1, 'd': 1}

使用defaultdict

from collections import defaultdict
some_data = ['a', '2', 2, 4, 5, '2', 'b', 4, 7, 'a', 5, 'd', 'a', 'z']

count_frq = defaultdict(int)
for item in some_data:
    count_frq[item] += 1
print count_frq

>>> defaultdict(<type 'int'>, {'a': 3, 2: 1, 'b': 1, 4: 2, 5: 2, 7: 1, '2': 2, 'z': 1, 'd': 1})

使用set和list

some_data = ['a', '2', 2, 4, 5, '2', 'b', 4, 7, 'a', 5, 'd', 'a', 'z']
count_set = set(some_data)
count_list = []
for item in count_set:
    count_list.append((item, some_data.count(item)))
print count_list

>>> [('a', 3), (2, 1), ('b', 1), (4, 2), (5, 2), (7, 1), ('2', 2), ('z', 1), ('d', 1)]

更Pythonic的方式，使用Counter

from collections import Counter
some_data = ['a', '2', 2, 4, 5, '2', 'b', 4, 7, 'a', 5, 'd', 'a', 'z']
print Counter(some_data)
>>> Counter({'a': 3, 4: 2, 5: 2, '2': 2, 2: 1, 'b': 1, 7: 1, 'z': 1, 'd': 1})

# pythonic 告诉我们 不要重复造轮子(狗头)

Counter类是Python2.7开始增加的，属于字典类的子类。

有三种方式来初始化

# 可迭代对象
Counter('success')
>>> Counter({'c': 2, 'e': 1, 's': 3, 'u': 1})

# 关键字参数
Counter(s=3, c=2, e=1, u=1)
>>>Counter({'c': 2, 'e': 1, 's': 3, 'u': 1})

# 字典
Counter({'s':3, 'c':2, 'u':1, 'e':1})
>>> Counter({'c': 2, 'e': 1, 's': 3, 'u': 1})

一些Counter类的方法

# 使用elements()方法来获取Counter中的Key值
list(Counter(some_data).elements())
>>> ['a', 'a', 'a', 2, 'b', 4, 4, 5, 5, 7, '2', '2', 'z', 'd']

# 使用most_common()方法，找出前N个出现频率最高的元素，以及对应出现次数
Counter(some_data).most_common(2)
>>> [('a', 3), (4, 2)]

使用ElementTree解析XML文件

xml.dom.minidom和xml.sax是python中解析XML最广为人知的两个模块。

dom是将整个XML文件加载到内存中，解析为一棵树，虽然使用简单，但是占用内存较多。

sax是基于事件驱动，虽不需要全部转入XML文件，但其处理过程相对复杂。

对于一般的XML处理，推荐ElementTree，它速度更快，消耗内存更少。

关于python 序列化

pickle估计是最通用的序列化模块，它还有一个C语言的实现cPickle。

相比pickle来说具有较好的性能，其速度大概是pickle的1000倍。

import cPickle as pickle

# 序列化
my_data = {'name': 'python', 'type': 'language', 'version': '2.7.5'}
fp = open('picklefile.dat', 'wb')
pickle.dump(my_data, fp)
fp.close()

# 反序列化
fp = open('picklefile.dat', 'rb')
out = pickle.load(fp)
fp.close()
print out

json是一种轻量级数据交换格式，它基于JaveScript编程语言的一个子集。

python中有许多对json格式支持的模块，如simplejson, cjson, yajl, ujson,和标准库JSON。

不过相对于pickle和cPickle，json的性能稍逊一筹。

使用traceback获取报错信息

对于用try except包装起来的异常，可以给用户很好的提示，但对于开发人员来说就不能详细的看到错误的完整栈信息。

g_list = ['a', 'b', 'c', 'd', 'e']
try:
	g_list[10]
except IndexError as ex
	print 'Sorry, Exception occured, you accessed an element out of range.'
	print ex

添加traceback后：

except IndexError as ex:
	print 'Sorry, Exception occured, you accessed an element out of range.'
	print ex
	traceback.print_exc()

程序运行后会输出完成的栈信息 ，包括调用顺序，异常放生的语句，错误类型等。
