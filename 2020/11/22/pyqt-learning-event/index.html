PySide Event事件

下面几个示例来说明事件的应用

  • 一些基本的事件注释 
# coding=utf-8

from PySide2.QtWidgets import *
from PySide2.QtGui import *
from PySide2.QtCore import *
import sys


class Window1(QWidget):

    def __init__(self):
        super(Window1, self).__init__()

        # 追踪鼠标移动
        # self.setMouseTracking(True)

    # 鼠标点击事件
    def mousePressEvent(self, event):
        # 鼠标点击时，打印点击位置(相对于窗口)
        print event.pos()

        # 鼠标点击时，打印点击位置(整个屏幕)
        print event.globalPos() # 可以在QMouseEvent里去寻找方法

        self.currentPos = self.pos()
        self.mouseOriPos = event.globalPos()

    # 鼠标移动事件
    def mouseMoveEvent(self, event):

        # 实现点击按住移动窗口功能
        self.mouseDesPos = event.globalPos()
        move = self.mouseDesPos - self.mouseOriPos
        self.move(self.currentPos + move)

    # 鼠标释放事件
    def mouseReleaseEvent(self, event):
        print 'mouse release'

    # 鼠标双击事件
    def mouseDoubleClickEvent(self, event):
        # 双击放大或还原默认窗口
        if self.isMaximized():
            self.showNormal()
        else:
            self.showMaximized()

    # 键盘按键事件
    def keyPressEvent(self, event):
        # 获取按下按键的值
        print event.text()

        # 设置按下ESC 关闭窗口
        if event.key() == Qt.Key_Escape:
            self.close()

        # 设置按下Ctrl + 回车键，发送消息
        if event.modifiers() == Qt.ControlModifier and event.key() == Qt.Key_Return:
            print 'send message.'

    # 释放键盘按键事件
    def keyReleaseEvent(self, event):
        print 'key release'

    # 鼠标进入窗口事件
    def enterEvent(self, event):
        print 'enter'

    # 鼠标离开窗口事件
    def leaveEvent(self, event):
        print 'leave'

    # 窗口关闭事件
    def closeEvent(self, event):
        print 'close'

app = QApplication(sys.argv)
widget = Window1()
widget.show()
app.exec_()

# coding=utf-8

from PySide2.QtWidgets import *
from PySide2.QtGui import *
from PySide2.QtCore import *
import sys


class Window1(QWidget):
    def __init__(self):
        super(Window1, self).__init__()
        self.setAcceptDrops(True)

    def dragEnterEvent(self, event):
        mimeData = event.mimeData()

        # 获取拖拽的数据类型
        print mimeData.formats()

        # 允许文件类型的拖拽
        if mimeData.hasUrls():
            event.accept()

    # 拖拽事件
    def dropEvent(self, event):
        print event.mimeData().urls()

        # 打印出所有拖拽进窗口的文件
        for url in event.mimeData().urls():
            print url.toLocalFile()


app = QApplication(sys.argv)
widget = Window1()
widget.show()
app.exec_()
