Nuke节点操作

获取节点

import nuke

# 已知节点的名称，使用toNode函数获取，返回一个节点对象。
nuke.toNode('Text1')

# 获取当前选择的单个节点，返回一个节点对象。
nuke.selectedNode()

# 获取当前选择的多个节点，返回一个列表。
nuke.selectedNodes()

# 获取工程所有节点
all_nodes = nuke.allNodes()

# 获取工程所有指定类型的节点
all_dot_nodes = nuke.allNodes("Dot")

获取，设置节点属性

# 获取cliptype的值
t = nuke.toNode('Text1')
print t['cliptype'].value()
>>> format

# 改变cliptype的值
t['cliptype'].setValue('bbox')

# 因为这个值是一个下拉菜单，所以可以传index参数
t['cliptype'].setValue(4)

# 设置Read文件路径,并自动set帧范围(对帧序列无效)
read_node['file'].fromUserText(video_file)

创建和删除节点

# 这种方式适合nuke界面打开的时候，代码执行后会跳转到节点的属性编辑窗。
nuke.createNode('Write')

# 这种方法适合没有打开nuke，在后台的一些操作。
nuke.nodes.Write()

# 创建节点，忽略弹出属性窗口
nuke.createNode("Blur", inpanel=False)

# 创建节点的时候，设置节点属性
nuke.nodes.Dot(xpos=100, ypos=100)

# 删除节点
nuke.delete(node)

# 删除工程指定类型的节点
for i in nuke.allNodes("Dot"):
  nuke.delete(i)

连接节点

t = nuke.toNode('Text1')
w = nuke.toNode('Write1')

# setInput的第一个参数是输入编号
# 第二个参数是需要连接的上游节点

w.setInput(0, t)

# 得到节点的input关系,返回节点对象
w.input(0)

root操作

# 获取nuke文件路径
nuke.root().name()

# 获取项目设置的起始帧
nuke.root().firstFrame()

使一个节点被选中

# 使一个节点被选择状态
node = nuke.toNode("Grade1")
node.setSelected(True)

渲染指定节点

# 参数分别是Write节点名，渲染起始帧，渲染结束帧，步长。

nuke.execute('Write1', 1001, 1010, 1)

获取指定路径的序列信息

file_path = r'/Users/Downloads/cam01'
result = nuke.getFileNameList(file_path)

# 可配合read节点加载序列帧文件

file = os.path.join(file_path, nuke.getFileNameList(file_path)[0])
nuke.createNode("Read", "file {%s}" % file)

获取设置节点位置

node = nuke.toNode("Grade1")
print node.xpos()
print node.ypos()

node.setXpos(0)
node.setYpos(20)
node.setXYpos(0, 20)

复制粘贴节点

# 将选择的节点复制到粘贴板
nuke.nodeCopy("%clipboard%")

# 将粘贴板的内容，粘贴到文件中
nuke.nodePaste("%clipboard%")

动画操作

# 设置为可key帧状态
node = nuke.toNode("Blur1")
knob = node("size")
knob.setAnimated()

# 给指定帧数key帧，第一个参数是数值，第二个参数是帧数
knob.setValueAt(55, 10)

# 比如size属性，有x和y两个值的时候，使用第三个参数来控制
knob.setValueAt(77, 100, 0)
knob.setValueAt(88, 100, 1)

# 判断一个属性是否有关键帧
print knob.isAnimated()

# 获取key帧动画
print knob.animations()

# 只获取size属性中，h的动画
print knob.animations(1)

# 获取size属性中，h的每一个关键帧和对应的数值
animation = knob.animation(1)
for key in animation.keys():
  print key.x, key.y
          # x为关键帧， y为对应的数值

# 获取非关键帧的数值
print animation.evaluate(60)

# 拷贝动画
nuke.toNode("Grade1")["white"].copyAnimations(knob.animations())

# 删除关键帧
for i in knob.animations():
  i.clear()
  knob.deleteAnimation(i)

文件操作

# 打开一个nuke文件
nuke.scriptOpen("$nuke_file_path")

# 打开一个新的空文件
nuke.scriptNew()

# 保存文件，如果括号里没有参数，则相当于ctrl+s
nuke.scriptSave("$nuke_file_path")

# 另存为
nuke.scriptSaveAs("$nuke_file_path")

# 关闭当前文件，回归到空文件状态
nuke.scriptClose()

# 将选择的节点，输出为一个nk文件
nuke.nodeCopy("$nuke_file_path")

# 将一个文件导入到当前工程
nuke.nodePaste("$nuke_file_path")
